import RPi.GPIO as GPIO
import time

GPIO.setmode(GPIO.BCM)
GPIO.setup(18,GPIO.IN)
GPIO.setup(17,GPIO.OUT, initial=0)

try:
	while True:
		data = GPIO.input(18)
		if data == 0:
			GPIO.output(17,GPIO.HIGH)
			print("Detected")
			time.sleep(1)
		else:
			GPIO.output(17,GPIO.LOW)
			print("Clear")
			time.sleep(1)
except KeyboardInterrupt:
	GPIO.cleanup(
