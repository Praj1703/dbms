import RPi.GPIO as GPIO
import time

LED = 17
SENSOR = 18

GPIO.setmode(GPIO.BCM)
GPIO.setup(SENSOR, GPIO.IN)
GPIO.setup(LED, GPIO.OUT, initial=0)

try:
    while True:
        if GPIO.input(SENSOR) == 0:
            GPIO.output(LED, GPIO.HIGH)
            print("Detected")
        else:
            GPIO.output(LED, GPIO.LOW)
            print("Clear")
        time.sleep(1)

except KeyboardInterrupt:
    GPIO.cleanup()
